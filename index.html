Setup the database 

            CREATE DATABASE CV; 
            CREATE USER ADMIN@localhost IDENTIDIED "PASS"; 
            GRANT ALL ON CV TO ADMIN@localhost; 
            CREATE TABLE Users ( 
                id INT AUTO_INCREMENT PRIMARY KEY UNIQUE,
                email VARCHAR(100),
                password VARCHAR(100)
            );
            CREATE TABLE Profile (
                id INT AUTO_INCREMENT PRIMARY KEY UNIQUE,
                users_id INT,
                year DATE,
                CONSTRAINT FOREIGN KEY (users_id) REFERENCES Users (id) 
                ON DELETE CASCADE ON UPDATE CASCADE
            );
